```ts
// src/rpc-agent/stacktrace.ts
// Variant of hookLog that includes a backtrace in the fire payload.
import "frida-il2cpp-bridge";
import { findClass, stringifyValue } from "../lib";

function frameToString(addr: NativePointer): string {
    try {
        const sym = DebugSymbol.fromAddress(addr);
        if (sym && sym.name) return `${sym.moduleName || "?"}!${sym.name}+0x${(addr.sub(sym.address as any)).toString(16)}`;
    } catch { /* fallthrough */ }
    return addr.toString();
}

export function hookLogWithStack(className: string, methodName: string, maxFrames = 12): Promise<void> {
    return new Promise((resolve, reject) => {
        Il2Cpp.perform(() => {
            try {
                const klass = findClass(className);
                if (!klass) { reject(new Error(`class ${className} not found`)); return; }
                const method = klass.tryMethod(methodName);
                if (!method) { reject(new Error(`method ${methodName} not found on ${className}`)); return; }
                method.implementation = function (this: any, ...args: any[]): any {
                    let ret: any, err: string | null = null;
                    try {
                        const self = this as Il2Cpp.Object;
                        ret = self.method(methodName).invoke(...args);
                    } catch (e) { err = String(e); }
                    const stack: string[] = [];
                    try {
                        const bt = Thread.backtrace(this.context, Backtracer.ACCURATE);
                        for (let i = 0; i < Math.min(bt.length, maxFrames); i++) {
                            stack.push(frameToString(bt[i]));
                        }
                    } catch { /* backtrace failed, leave stack empty */ }
                    send({
                        type: "hook",
                        cls: className,
                        method: methodName,
                        self: String((this as any)?.handle ?? "static"),
                        args: args.map((a: any) => stringifyValue(a)),
                        retval: stringifyValue(ret),
                        error: err,
                        stack,
                    });
                    if (err) throw new Error(err);
                    return ret;
                };
                console.log(`[hook+stack] installed on ${className}.${methodName}`);
                resolve();
            } catch (e) { reject(e); }
        });
    });
}
```

Add to `rpc-methods.ts`: `import * as stacktraceRpc from "./stacktrace"; ...stacktraceRpc`.

**UI in hookpatch.ts**: next to the existing `log hook` button, add a `hook+stack` button. Alternative: a single checkbox `[ ] capture stack` — when checked, clicking `log hook` calls `hookLogWithStack` instead of `hook`. Use the checkbox approach (simpler UX).

Add an `<input class="input" placeholder="regex filter…">` above the `#log` container. When the user types, re-apply a filter:

```ts
let filterRegex: RegExp | null = null;
const filterInput = document.createElement("input");
filterInput.className = "input";
filterInput.placeholder = "regex filter (empty = show all)";
filterInput.style.cssText = "margin-bottom: var(--s-2)";
filterInput.addEventListener("input", () => {
    const v = filterInput.value.trim();
    try { filterRegex = v ? new RegExp(v, "i") : null; filterInput.style.borderColor = ""; }
    catch { filterRegex = null; filterInput.style.borderColor = "var(--err)"; }
    applyFilter();
});
container.parentElement?.insertBefore(filterInput, container);

function applyFilter(): void {
    const lines = container.querySelectorAll<HTMLElement>(".line");
    for (const line of lines) {
        const txt = line.textContent ?? "";
        const classMatches = showFilters[line.dataset["cls"] as "log" | "hook" | "rpc"] !== false;
        const regexMatches = !filterRegex || filterRegex.test(txt);
        line.style.display = (classMatches && regexMatches) ? "" : "none";
    }
}
```

Also apply the filter to new lines in `appendLine` (check regex, set display accordingly before inserting).

**Persistence (`saveDump`):**

```js
function saveDump(content, meta = {}) {
    ensureDir(path.join(DATA_DIR, "dumps"));
    const ts = new Date().toISOString().replace(/[:.]/g, "-");
    const name = (meta.name || "dump").replace(/[^a-zA-Z0-9._-]/g, "_").slice(0, 60);
    const ext = (meta.ext || "md").replace(/[^a-z]/gi, "") || "md";
    const file = path.join(DATA_DIR, "dumps", `${ts}-${name}.${ext}`);
    fs.writeFileSync(file, content, "utf8");
    return { path: file, size: Buffer.byteLength(content, "utf8"), name };
}
```

**Server route:**

```js
POST: {
    ...
    "/api/dumps": async (req, res) => {
        const body = JSON.parse(await readBody(req));
        sendJson(res, 200, persistence.saveDump(body.content || "", body.meta || {}));
    },
}
```

**UI (search.ts)**: next to the `FULL`/`STATICS` buttons, a `DUMP TO FILE` button. Clicking it performs a `dumpClass(className)` via RPC, captures the logged output into a string, and POSTs to `/api/dumps` with `content` = captured markdown and `meta.name` = className. On success, show a brief toast "saved → path".

Implementation challenge: `dumpClass` currently outputs via `console.log` which goes to the event stream, not a buffer. To capture, we need to either:
- A. Wrap `rpcCall("dumpClass", args)` — but it returns void, all text is in the log stream.
- B. Add a new RPC `dumpClassAsString(className)` that returns the markdown directly.

**Recommend approach B** — add `dumpClassAsString` to `src/rpc-agent/search.ts`. Much cleaner than log capture.

```ts
// in search.ts, alongside existing dumpClass
export function dumpClassAsString(name: string): Promise<string> {
    return new Promise((resolve, reject) => {
        Il2Cpp.perform(() => {
            const k = findClass(name);
            if (!k) { reject(new Error(`class ${name} not found`)); return; }
            const lines: string[] = [`# ${name}`, ""];
            // Mirror the existing dumpClass formatting from src/lib/dump.ts
            lines.push(`**Fields (${k.fields.length})**`, "");
            for (const f of k.fields) {
                lines.push(`- ${f.isStatic ? "static " : ""}${f.type.name} ${f.name}`);
            }
            lines.push("", `**Methods (${k.methods.length})**`, "");
            for (const m of k.methods) {
                const kind = m.isStatic ? "static " : "";
                const params = m.parameters.map(p => `${p.type.name} ${p.name}`).join(", ");
                lines.push(`- ${kind}${m.returnType.name} ${m.name}(${params})`);
            }
            resolve(lines.join("\n"));
        });
    });
}
```

Similarly `dumpInstanceAsString(className)` in instance-ops module.

UI flow for search panel's DUMP TO FILE:
1. Get `className` from the dump input.
2. `const md = await rpcCall<string>("dumpClassAsString", [className])`
3. `await fetch("/api/dumps", {method:"POST", headers:{"Content-Type":"application/json"}, body: JSON.stringify({content: md, meta: {name: className, ext: "md"}})})`
4. Parse response, log `[dump] saved → ${path}` to event log.

## Task 5: Keybinds

```ts
document.addEventListener("keydown", (e) => {
    // Ctrl+K — clear log
    if (e.ctrlKey && !e.shiftKey && e.key.toLowerCase() === "k") {
        e.preventDefault();
        document.getElementById("log")!.innerHTML = "";
        return;
    }
    // Ctrl+Shift+C — copy session (trigger existing button)
    if (e.ctrlKey && e.shiftKey && e.key.toLowerCase() === "c") {
        e.preventDefault();
        const btn = [...document.querySelectorAll("button")].find(b => b.textContent?.includes("COPY SESSION")) as HTMLButtonElement | undefined;
        btn?.click();
        return;
    }
    // Escape — dismiss toast (bookmark auto-offer)
    if (e.key === "Escape") {
        const toast = document.querySelector(".bookmark-toast");
        if (toast) toast.remove();
    }
});
```

## Self-Review

- Task 3 split into 3 commits because each layer (RPC, server persistence, UI) is independent.
- Stack trace implementation uses Frida's `Thread.backtrace` with `Backtracer.ACCURATE` which works even for IL2CPP methods since the hook trampolines preserve stack. `DebugSymbol.fromAddress` may fail for JIT'd IL2CPP code — graceful fallback is the raw address.
- Filter input state is intentionally NOT persisted across reloads — users expect fresh filter on reload.
- Keybinds intentionally avoid single-letter (clashes with typing in inputs).

File sizes: `stacktrace.ts` ~60; `persistence.js` grows by ~12; other patches are small edits (< 30 lines each).
